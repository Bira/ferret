179:         def obtain(lock_timeout = 1) 
180:           return true if FSDirectory.locks_disabled?
181:           MAX_ATTEMPTS.times do
182:             begin
183:               # create a file if none exists. If one already exists
184:               # then someone beat us to the lock so return false
185:               File.open(@lock_file, File::WRONLY|File::EXCL|File::CREAT) {|f|}
186:               return true
187:             rescue SystemCallError
188:               # lock was not obtained so sleep for timeout then try again.
189:               sleep(lock_timeout)
190:             end
191:           end
